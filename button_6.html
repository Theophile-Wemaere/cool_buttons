Neon button 
          
a {
  color: #03e9f4;
  text-decoration: none; /* Supprime le soulignement dû au lien (balise a) */
  position: relative;
  text-transform: uppercase; /* Tout mettre en majuscule*/
  padding: 25px 30px;
  letter-spacing: 4px; /* Lettre détachée*/
  transition: 0.5s all;
  overflow: hidden;
}

a:hover {
  background-color: #03e9f4;
  color: #050801;
}

a span {
  position: absolute;
}

/* On va styliser chacune des balises span une par une*/

a span.span1 {
  /* on place la première balise span en haut du bouton */
  top: 0;
  left: 0;
  width: 100%;
  height: 2px;
  background: linear-gradient(90deg, transparent, #03e9f4);
  animation: animate1 1s linear infinite;
}

@keyframes animate1 {
  /* Création de la première animation */
  0% {
    left: -100%;
  }
  50%,
  100% {
    left: 100%;
  }
}
