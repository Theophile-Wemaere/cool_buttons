Comment faire des boutons stylés en CSS ? 
          
button {
  font-family: system-ui, sans-serif;
  font-weight: bold;
  font-size: 2rem;
  cursor: pointer;
  border: none;
  margin: .1em;
}

button:hover,
  button:focus-visible {
  --_p: 0px;
  outline-color: var(--c);
  outline-offset: .05em;
}

button:active {
  background: var(--c);
  color: #fff;
}

button {
  --b: 3px;   /* border thickness */
  --s: .50em; /* size of the corner */
  --c: #BD5532;
  
  padding: calc(.50em + var(--s)) calc(.3em + var(--s));
  padding-left: 30px;
  padding-right: 30px;
  color: var(--c);
  --_p: var(--s);
  background:
    conic-gradient(from 90deg at var(--b) var(--b),#0000 90deg,var(--c) 0)
    var(--_p) var(--_p)/calc(100% - var(--b) - 2*var(--_p)) calc(100% - var(--b) - 2*var(--_p));
  transition: .3s linear, color 0s, background-color 0s;
  outline: var(--b) solid #0000;
  outline-offset: .2em;
}
